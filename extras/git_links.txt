git init // inicia a linha do tempo

git add nome.php //adicionar mudanças no git

git commit -m "descrição" //criar pontos na linha do tempo

git log // mostra historico commits

git status // estado de desenvolvimento do git

git show //apresente derterminado ponto na historia

git branch // gerencia novas linas do tempo

git checkout //manipula linhas do tempo

git merge // une linhas do tempo

git push -u origin main //joga pra nuvem 

git pull //baixa att na nuvem

git clone https://github.com/rameck88/ptcc.git //link do projeto
